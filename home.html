Python SDK v2.4 
from supramath import Client

client = Client(api_key="sm_live_XXXX")

response = client.solve(
    expression="d/dx (e^(2x) * cos(x))",
    variables={"x": 0},
    format=["symbolic", "numeric", "steps"],
)

print(response.symbolic)
# 2e^(2x)cos(x) - e^(2x)sin(x)
